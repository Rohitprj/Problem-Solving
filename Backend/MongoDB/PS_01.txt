Topic 1 — $match
Question 1

Find all MERN students.

Answer
db.students.aggregate([
  {
    $match: {
      course: "MERN"
    }
  }
]);

Question 2

Find all Java students.

Answer
db.students.aggregate([
  {
    $match: {
      course: "Java"
    }
  }
]);

Question 3

Find students whose marks are greater than 80.

Answer
db.students.aggregate([
  {
    $match: {
      marks: { $gt: 80 }
    }
  }
]);

Topic 6 — $group
Question 14

Show unique courses.

Answer
db.students.aggregate([
  {
    $group: {
      _id: "$course"
    }
  }
]);

Question 15

Count students in each course.

Answer
db.students.aggregate([
  {
    $group: {
      _id: "$course",
      totalStudents: {
        $sum: 1
      }
    }
  }
]);
Output
{ "_id": "MERN", "totalStudents": 2 }
{ "_id": "Java", "totalStudents": 2 }
{ "_id": "Python", "totalStudents": 1 }

Question 16

Count total students.

Answer
db.students.aggregate([
  {
    $group: {
      _id: null,
      totalStudents: {
        $sum: 1
      }
    }
  }
]);

Topic 7 — $sum
Question 17

Find total fees of all students.

Answer
db.students.aggregate([

  }
]);
